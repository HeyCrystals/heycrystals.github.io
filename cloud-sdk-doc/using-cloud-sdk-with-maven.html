使用maven

cloud-sdk最新版本jar包已经上传到卓锐maven私服上,可以在项目中配置私服引用

私服访问地址:http://192.168.11.110:8081/nexus/#welcome,

maven使用配置如下：

<!-- 配置maven私服地址-->
<repositories>
    <repository>
        <id>nexus</id>
        <name>drore Team nexus</name>
        <url>http://192.168.11.110:8081/nexus/content/groups/public</url>
    </repository>
</repositories>
<pluginRepositories>
    <pluginRepository>
        <id>nexus</id>
        <name>drore Team nexus</name>
        <url>http://192.168.11.110:8081/nexus/content/groups/public</url>
    </pluginRepository>
</pluginRepositories>
<dependencies>
    <!--  cloud-sdk: Java to open-implement-kit conversion -->
<!-- 添加maven依赖-->
    <dependency>
        <groupId>com.drore.cloud</groupId>
        <artifactId>cloud-sdk</artifactId>
        <version>0.4.3</version>
    </dependency>
</dependencies>
