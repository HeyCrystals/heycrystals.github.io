sdk代码使用示例：

@Override
    public Pagination<Map> myCouponList(Integer current_page, Integer page_size) {
        MemberInfo memberInfo=(MemberInfo) ThreadLocalHolder.getSession().getAttribute(LocalConstant.SESSION_CURRENT_USER);
        RequestExample requestExample=new RequestExample(page_size,current_page);
        RequestExample.Criteria criteria=requestExample.create();
        //主表查询
        criteria.getMust().add(requestExample.createParam().addTerm("user_id",memberInfo.getId()));
        //创建关联表
        Relation couponRelation=new Relation("coupon_info","coupon_info","id","coupon_id");
        //关联表筛选条件
        couponRelation.getWhere().getMust().add(requestExample.createParam().addTerm("test","1"));
        //addRelation
        requestExample.addRelations(couponRelation);
        //排序
        requestExample.addSort("create_time","desc");
        Pagination<Map> mapPagination=run.queryListByExample("coupon_member_info",requestExample);
        return mapPagination;
    }
