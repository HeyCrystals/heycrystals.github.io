使用CloudQueryRunner(推荐)

@RestController
@RequestMapping("/convert")
public class ConvertController {
 static final String EDIT="E";
 static final String ISSUE="I";
 @Autowired
 private CloudQueryRunner run;

@PostMapping(value = "/list")
public RestMessage queryList(@RequestParam(value = "current_page",required = false,defaultValue = "1")Integer current_page, @RequestParam(value = "page_size",required = false,defaultValue = "10")Integer page_size, @RequestParam(value = "name",required = false)String name, @RequestParam(value = "type_id",required = false)String type_id){
 RequestExample requestExample=new RequestExample(page_size,current_page);
 //关联分类表
 Relation r=new Relation("bi_type_info","bi_type_info","id","type_id");
 //关联模板表
 Relation modelRelation=new Relation("bi_model","bi_model","id","model_id");
 requestExample.setAlias("bi_convert");
 requestExample.addRelations(r);
 requestExample.addRelations(modelRelation);
 RequestExample.Criteria criteria=requestExample.create();
 if (StringUtils.isNotBlank(name)){
 criteria.getMust().add(requestExample.createParam().addFuzzy("name",name));
 }
 if (StringUtils.isNotBlank(type_id)){
 criteria.getMust().add(requestExample.createParam().addTerm("type_id",type_id));
 }
 return new RestMessage(run.queryListByExample("bi_convert",requestExample));
}
}
